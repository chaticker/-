```python
from itertools import permutations

def solution(numbers):
    answer = 0
    a_list = []
    
    #숫자 조합을 a_list에 넣기
    for i in range(len(numbers)):
        a = list(permutations(numbers, i+1))
        for j in a:
            a_list.append(int(''.join(j)))
            #print(ex) -> [1, 7, 17, 71]
    
    #중복 제거 후 소수인지 판별
    for i in set(a_list):
        b = 0
        for j in range(1, i+1):
            if i % j == 0 and i != 0:
                b += 1
            if b >= 3:
                break
        if b == 2:
            answer += 1
    return answer
```
