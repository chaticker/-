```python
def solution(progresses, speeds):
    answer = []
    p = 0 #현재 진행중인 곳을 의미 (왼쪽부터 차례대로 움직이기 때문에 가능)
    while len(progresses) != p:
        cnt = 0
        for i in range(p, len(progresses)):
            progresses[i] += speeds[i]
        
        if progresses[p] >= 100: #현재 진행인 p가 100이상인 경우 완료된 건 빼고
            for i in range(p, len(progresses)):
                if progresses[i] >= 100:
                    cnt +=1
                else:
                    break
            
            answer.append(cnt) #answer에 더해줌
            p += cnt #p의 위치를 성공한 곳까지 움직임
    return answer
```
